 \section{Les triangles semblables}
 
 
 \subsection{Définition}
 
 \df{
 Deux triangles sont semblables si tous les angles de l'un sont égaux aux angles de l'autre. 
 }\\
 
 
 \underline{\textbf{Exemples :}}\\
 
\begin{center}
 \includegraphics[scale=0.8]{exsemblable12.eps}  
\end{center}
 
 Les triangles $A_{1}B_{1}C_{1}$ et $A_{2}B_{2}C_{2}$ sont dits semblables car $ \widehat{A_{1}B_{1}C_{1}}= \widehat{A_{2}B_{2}C_{2}}$, $ \widehat{A_{1}C_{1}B_{1}}= \widehat{A_{2}C_{2}B_{2}}$ et $ \widehat{C_{1}A_{1}B_{1}}= \widehat{C_{2}A_{2}B_{2}}$.\\
 
 
 \underline{\textbf{Remarque :}} Pour que deux triangles soient semblables, il suffit que deux angles de l'un des triangles soient égaux à deux angles de 
l'autre triangle.\\
 
\begin{Exoapp}
 
Prouver que les deux triangles ci-dessous sont semblables.\\

\bmul{2}

 \includegraphics[scale=0.8]{exsemblable3.eps}  

\columnbreak

\noindent \reponse[5]

\emul

\end{Exoapp}

\section{Cas particulier - Théorème de Thalès}


\subsection{Le théorème}

\begin{center}
\includegraphics[scale=0.8]{thalesfig.eps}  \includegraphics[scale=1]{figurethales.eps} 
 \end{center} 
 
 \vspace*{0.3cm}


\thm{
Soient ABC un triangle quelconque non aplati.\\
Si les droites (AB) et (AC) sont sécantes en A et si la droite (MN) est parallèle à la droite (BC).\\
Alors on a l'égalité suivante :\\

\begin{center}
$\dfrac{AM}{AB}=\dfrac{AN}{AC}=\dfrac{MN}{BC}$
\end{center}
}\\
\vspace*{0.3cm}


\underline{\textbf{Exemple}}\\

On suppose que les droites  (LB) et (IC) sont parallèles.\\

 \hspace*{6cm} \includegraphics[scale=0.9]{thales5.eps} \\





\subsection{Application du Théorème de Thalès}
\vspace*{0.3cm}


\underline{\textbf{Objectif}} \textbf{: Le théorème de Thalès permet de calculer des longueurs de segments.} \\

\vspace*{0.3cm}


\underline{Énoncé :}
\bmul{2}
On considère un triangle ABC tel que :\\

AD = 3,7 dm ; AB = 5,3 dm ; DE = 4,1 dm et AC = 8,2 dm. \\

Les droites (BC) et (DE) sont parallèles.\\

\textbf{Calculer BC et AE au dixième près.}\\

\columnbreak

\includegraphics[scale=0.65]{figurethales2.eps} 

\emul

\vspace*{0.3cm}

\underline{\textbf{Résolution :}} \\

Dans le triangle ABC :
\bi
\item Les droites (AB) et (AC) sont sécantes en A. Les points A, D, B et A, E, C sont alignés dans le même ordre.
\item (ED) $\slash\slash$ (BC)
\ei

D'après le théorème de Thalès, on a:

\begin{center}
$\dfrac{AD}{AB}=\dfrac{AE}{AC}=\dfrac{DE}{BC}$\\
\end{center}
 
On remplace : \hspace*{1cm} $\dfrac{3,7}{5,3}=\dfrac{AE}{8,2}=\dfrac{4,1}{BC}$\\

\bmul{2}

\textbf{Calcul de BC :}\\

$\dfrac{3,7}{5,3}=\dfrac{4,1}{BC}$ donc $BC = \dfrac{5,3 \times 4,1}{3,7}$\\

BC $\approx$ 5,9  dm 


\columnbreak

\textbf{Calcul de AE :}\\

$\dfrac{3,7}{5,3}= \dfrac{AE}{8,2}$ donc $AE = \dfrac{3,7 \times 8,2}{5,3}$\\

AE $\approx$ 5,7  dm 


\emul
