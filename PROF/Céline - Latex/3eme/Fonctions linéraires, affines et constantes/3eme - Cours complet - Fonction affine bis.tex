\section{Introduction}

\underline{\textbf{{\large Enoncé :}}}\\

Un club multi-sports propose à sa clientèle de choisir entre les trois formules suivantes :\\
\textbf{Formule A} : 10 euros par séance.\\
\textbf{Formule B} : Un forfait annuel de 150 \euro  auquel s'ajoute une participation de 5 \euro par séance.\\
\textbf{Formule C} : Un forfait annuel de 500 \euro permettant l'accès illimité aux séances.\\

\initq \q Calculer pour chaque formule la dépense annuelle pour :
15 séances ;  40 séances ; 50 séances ; 75 séances ; 90 séances.
Dans chaque cas, quelle est la formule la plus intéressante ?\\

\q Soit x le nombre de séances pendant une année. Exprimer en fonction de x la dépense annuelle pour chaque formule.\\


\q \initqa\qa Pour chaque formule, représenter sur un même graphique la dépense annuelle en fonction du nombre d'entrées.\\
\qa Déterminer graphiquement la formule la plus avantageuse en fonction du nombre de séances.\\

\textbf{{\large \underline{Résolution : }}}\\

\initq \q \color{red} Pour 15 séances : 

\bmul{3}

FORMULE A :\\
$15 \times 10 = 150 $ euros

\columnbreak

FORMULE B :\\
$150 + 15 \times 5 = 225 $ euros

\section{Fonctions affines}

\subsection{Définition}

\vspace*{0.5cm}

\df{
On dit qu'une fonction f est affine s'il existe deux nombres a
et b tel que f : $f:x\longmapsto ax + b$.\\
Le nombre \textbf{a} est appelé \textbf{coefficient directeur} de la fonction f et le nombre \textbf{b} est appelé \textbf{ordonnée à l'origine}.\\
}

\vspace*{0.5cm}

\rem
\bi \item \textbf{Une fonction linéaire} est une fonction affine où \textcolor{red}{$b=0$}.\\
\item \textbf{Une fonction constante} est une fonction affine où \textcolor{red}{$a=0$}. \\
\ei

\vspace*{0.5cm}

\underline{\textbf{Exemples :}}

\begin{center}
{\renewcommand{\arraystretch}{1.5}
\begin{tabular}{|l|l|l|}
	\hline
	\textbf{Fonction} & \textbf{Linéaire ? Constante ? Affine ?} & \textbf{Coefficients ?}\\
	\hline \hline
	$f:x\longmapsto 5x$ & \textcolor{red}{linéaire donc affine} & \textcolor{red}{$a=5  et b=0$}\\
	\hline
	$g:x\longmapsto 5x+2$ &\textcolor{red}{ affine} & \textcolor{red}{$a=5  et b=2$}\\
	\hline
	$h:x\longmapsto 8$ &\textcolor{red}{constante donc affine} & \textcolor{red}{$a=0  et b=8$}\\
	\hline
	$i:x\longmapsto \dfrac{x-8}{3}$ & \textcolor{red}{ affine} & \textcolor{red}{$a=\dfrac{1}{3}  et b=\dfrac{8}{3}$}\\
	\hline
	$j:x\longmapsto x^2$ & X &  X\\
	\hline
\end{tabular}}\end{center}

\vspace*{1.5cm}

\begin{Exoapp}

\textbf{Calculer des images connaissant les antécédents.}\\

On donne $f:x \longmapsto -4x+2$ et $g:x \longmapsto \displaystyle{\frac{x-1}{2}}$ . Calculer $f(3)$, $g(-1)$ et $g(1)$.

\color{red} 

\bmul{3}

- Calcul de  $f(3)$:\\

$f(3) = -4 \times 3 + 2 $\\

$f(3) = -12+ 2 $\\

$f(3) = -10 $\\

\columnbreak

- Calcul de  $g(-1)$:\\

$g(-1) = \dfrac{-1-1}{2} $\\

$g(-1) = \dfrac{-2}{2} $\\

$g(-1) = -1 $\\

\columnbreak

- Calcul de  $g(1)$:\\

$g(1) = \dfrac{1-1}{2} $\\

$g(1) = \dfrac{0}{2} $\\

$g(1) = 0 $\\

\emul


\color{black} 

\end{Exoapp}

\newpage

\begin{Exoapp}

\textbf{Déterminer des antécédents connaissant les images.}\\

On donne la fonction $f:x\longmapsto -2x+3$. Déterminer les antécédents de -5 et de 3.

\color{red} 

Pour calculer le ou les antécédents de -5 par la fonction $f$, on résout l'équation \hspace*{0.5cm} $-2x+3=-5$.\\

$$-2x+3=-5$$
$$-2x=-5-3$$
$$-2x=-8$$
$$\dfrac{-2x}{-2}=\dfrac{-8}{-2}$$
$$x=4$$

L'antécédent de -5 par la fonction f est 4.\\

Pour calculer le ou les antécédents de 3 par la fonction $f$, on résout l'équation \hspace*{0.5cm} $-2x+3=3$.\\

$$-2x+3=3$$
$$-2x=-3-3$$
$$-2x=0$$
$$x=0$$

L'antécédent de 3 par la fonction f est 0.\\

\color{black} 


\end{Exoapp}







\subsection{Propriétés}

\prop{
Soient f une fonction affine, $x_{1}$ et $x_{2}$ deux nombres.\\

Si $x1 \neq x2$ alors \hspace*{0.4cm} $ a = \dfrac{f(x_{2}) - f(x_{1})}{x_{2}-x_{1}}$


}\\

\vspace*{0.5cm}

\underline{\textbf{Exemple :}}\\

$\rightarrow$ Déterminer la fonctions affine telle que f(1) = 2 et f(3) = -4.\\

\underline{Étape 1}: Calcul du coefficient a.\\
Pour trouver le coefficient a, nous allons utiliser la propriété ci-dessus. On a f(1) = 2 et f(3) = -4.\\

Ainsi, $a = \dfrac{f(3) -f(1)}{3-1}$ \hspace*{1cm} On remplace par les valeurs :\\

$a = \dfrac{-4-2}{3-1}$\\

$a = \dfrac{-6}{2}$ \hspace*{1cm}  $a = -3$\\

Dès lors, on obtient que, pour tout réel x, $f(x)=-3x+b$.

\newpage

\underline{Étape 2 }: Calcul du coefficient b.\\

Pour cela, il faut utiliser une des 2 égalités de l'énoncé. Prenons, f(1)=2. Cela signifie de l'image de 1 est 2 par la fonction $f(x)=-3x+b$.\\

$f(1)=2$ et $f(x)=-3x+b$ implique que :\\

$-3 \times 1 + b = 2$\\

$-3+b=2$\\ Il n'y a plus qu'à résoudre l'équation.\\

$b=2+3$\\

$b=5$ \\   




\underline{Étape 3 }: Expression de la fonction f.\\

L'expression de la fonction affine f est donc   $f(x)=-3x+5$.\\

\begin{Exoapp}

\textbf{Déterminer une fonction affine à l'aide de deux nombres et de leur image.}\\

Déterminer la fonction affine f telle que f (1) = 3 et f (-2) = 0.\\

\color{red} 

\underline{Étape 1}: Calcul du coefficient a.\\
Pour trouver le coefficient a, nous allons utiliser la propriété ci-dessus. On a f (1) = 3 et f (-2) = 0.\\

Ainsi, $a = \dfrac{f(-2) -f(1)}{-2-1}$ \hspace*{1cm} On remplace par les valeurs :\\

$a = \dfrac{0-3}{-2-1}$\\

$a = \dfrac{-3}{-3}$ \hspace*{1cm}  $a = 1$\\

Dès lors, on obtient que, pour tout réel x, $f(x)=1x+b$.

\newpage

\underline{Étape 2 }: Calcul du coefficient b.\\

Pour cela, il faut utiliser une des 2 égalités de l'énoncé. Prenons, f(1)=2. Cela signifie de l'image de 1 est 2 par la fonction $f(x)=1x+b$.\\

$f(1)=3$ et $f(x)=1x+b$ implique que :\\

$1 \times 1 + b = 3$\\

$1+b=3$\\ Il n'y a plus qu'à résoudre l'équation.\\

$b=3-1$\\

$b=2$ \\   




\underline{Étape 3 }: Expression de la fonction f.\\

L'expression de la fonction affine f est donc   $f(x)=1x+2$.\\


\color{black} 


\end{Exoapp}
