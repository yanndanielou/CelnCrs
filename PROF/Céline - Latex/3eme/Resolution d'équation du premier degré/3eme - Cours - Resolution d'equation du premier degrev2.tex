\begin{itemize}[label=$\hookrightarrow$]

\end{itemize}

\subsection{Définition}

\df{
Une équation est \textbf{une égalité} dans laquelle figure \textbf{un nombre inconnu}, désigné en général par une lettre qui est appelée l'inconnue.
}\\

\underline{\textbf{Exemple} }:\\


2x - 11 = 7 - x est une équation dans laquelle l'inconnue est désignée par $x$. \\

\begin{center}
$\underbrace{2x-11}_{\text{Premier membre}} = \underbrace{7-x}_{\text{Second membre}}$\\
\end{center}

\vspace*{0.5cm}
\subsection{Comment savoir si un nombre est solution d'une équation ?}

\df{
\textbf{Résoudre une équation d'inconnue $x$}, c'est trouver toutes les valeurs possibles du nombre $x$ (si elles existent) qui vérifient l'égalité, c'est-à-dire que l'égalité soit vraie.\\
Chacune de ces valeurs est \textbf{une solution de l'équation}.
}\\



\textbf{$\blacktriangleright$ Le nombre 3 est-il solution de l'équation $2x - 11 = 7 - x$ ?}\\

\bmul{2}
D'une part, $2 \times 3 -11 = \underline{- 5}$

\columnbreak

D'autre part, $7- 3 = \underline{4}$
\emul

L'égalité n'est donc pas vérifiée pour $x=3$.\\

\underline{Conclusion }: Le nombre 3 n'est pas solution de l'équation.\\

\vspace*{0.5cm}

\textbf{$\blacktriangleright$ Le nombre 6 est-il solution de l'équation $2x - 11 = 7 - x$ ?}

\bmul{2}
D'une part, $2 \times 6 -11 = \underline{1}$

\columnbreak

D'autre part, $7- 6 = \underline{1}$
\emul

L'égalité est donc vérifiée pour $x=6$\\

\underline{Conclusion }: Le nombre 6 est une solution de l'équation.\\

\vspace*{0.5cm}

\end{Exoapp}




\section{Méthodes de résolution d'équation}




 
\subsection{Première propriété}

\vspace*{0.5cm}

\prop{
Une égalité reste vraie si l'on \textbf{additionne} (ou l'on \textbf{soustrait}) le même nombre à \textbf{chacun de ses membres}.\\
}\\
\newpage

$\rightarrow$ \textbf{Résoudre l'équation $3+x=7$.}\\


\begin{align*}
3+x &= 7 \\
\cancel{3} + x \textcolor{red}{- \cancel{ 3}}  &=  7 \textcolor{red}{- 3}\\
x&=7-3\\
x&=4
\end{align*}





\textbf{La solution de l'équation $3 + x = 7$ est le nombre 4. \hspace*{0.75cm} On notera alors $\mathscr{S}$ = $\left\{4\right\}$.}\\

\vspace*{0.5cm}


\begin{Exoapp}
Résoudre les équations suivantes :\\

\bmul{2}

\begin{center}
$x-2=11$\\
\end{center}
\reponse[2]


\columnbreak

\begin{center}
$9+x=-44$\\
\end{center}
\reponse[2]

\emul




\end{Exoapp}




\subsection{Deuxième propriété}

\vspace*{0.5cm}

\prop{
Une égalité reste vraie si l'on \textbf{multiplie} (ou l'on \textbf{divise}) le même nombre non nul (c'est-à-dire différent de 0) à \textbf{chacun de ses membres}.
}\\

$\rightarrow$ \textbf{Résoudre l'équation $5x = 30$}.\\




\begin{align*}
5x &= 30\\
\dfrac{\cancel{5}x}{\textcolor{red}{{\cancel{5}}}} &= \dfrac{30}{\textcolor{red}{5}}\\
x &= \dfrac{30}{5}\\
 x &= 6\\
\end{align*}

\textbf{La solution de l'équation $5x = 30$ est le nombre 6. \hspace*{0.75cm} On notera alors $\mathscr{S}$ = $\left\{6\right\}$.}\\

\vspace*{0.5cm}

\subsection{Méthode de résolution des équations  avec des inconnues dans les 2 membres }

\vspace*{0.3cm}

$\rightarrow$ \textbf{Résoudre l'équation $7x - 2 = 6 + 5x$}.\\

\bmul{2}



$$7x - 2 = 6 + 5x$$

$$7x - 2 \textcolor{red}{- 5x} = 6 + 5x \textcolor{red}{- 5x}$$

$$2x - 2 = 6$$

$$2x - 2 \textcolor{red}{+ 2 }= 6 \textcolor{red}{+ 2}$$

$$2x = 8$$

$$\dfrac{\cancel{2}x}{\textcolor{red}{\cancel{2}}} = \dfrac{8}{\textcolor{red}{2}}$$

$$\fbox{x=4}$$





\columnbreak

\begin{center}
\textbf{\underline{Les étapes :}}
\end{center}

$\rightarrow$ On commence par isoler l'inconnue dans un des deux membres.\\

$\rightarrow$ On regroupe ensuite les nombres sans $x$ dans l'autre
membre.\\

$\rightarrow$ On utilise alors les méthodes de résolution vues juste avant.\\

$\rightarrow$ On n'oublie pas de donner la ou les solution(s).\\

\emul



\textbf{La solution de l'équation $7x - 2 = 6 + 5x$ est le nombre 4. \hspace*{0.75cm} $\mathscr{S}$ = $\left\{4\right\}$.}\\

\section{Mise en équation}

\subsection{Énigme 1}

\underline{\textbf{Énoncé}} : Alexandra et Thomas choisissent un même nombre. Alexandra multiplie ce nombre par 5 et ajoute 14 au résultat.\\
Thomas ajoute 29 au nombre choisi. Ils trouvent le même résultat.\\
\textbf{Quel est le nombre qu'ils ont choisi au départ ?}\\

 

\underline{\textbf{Résolution}} : \\

\textbf{\initq \q} \textbf{Choisir une inconnue et la décrire} : \hspace*{1cm}  On choisi $x$ le nombre choisi au départ.\\ 

\textbf{\q}  \textbf{Traduire le problème par une équation:}\hspace*{1cm} $5x+14=x+29$\\

\textbf{\q}  \textbf{Résoudre l'équation:}\\

\begin{align*}
5x+14&=x+29\\
5x+14 \textcolor{red}{-x} &= \cancel{x}+29 \textcolor{red}{-\cancel{x}}\\
4x+14&=29\\
4x +\cancel{14}\textcolor{red}{-\cancel{ 14}}&=29\textcolor{red}{-14} \\
4x&=15\\
\dfrac{4x}{\textcolor{red}{4}}&=\dfrac{15}{\textcolor{red}{4}}\\
x&=3,75
\end{align*}


\textbf{\q}  \textbf{Vérifier et conclure. }  \\

D'une part: $5 \times 3,75 +14 = 18,75 + 14 = 32,75$ \hspace*{1cm} D'autre part : $3,75 +29 = 32,75$\\
\textbf{Le nombre choisi est bien 3,75.}

\subsection{Énigme 2}

\underline{\textbf{Énoncé}} : Mathieu a 2 billes de plus que Pierre mais 3 fois moins de billes que Bryan.\\
Ils ont à eux trois 53 billes.\\
\textbf{Combien ont-ils de billes chacun?}\\

\vspace*{0.5cm}
\underline{\textbf{Résolution}} : \\

\initq
\textbf{ \q} \textbf{Choisir une inconnue et la décrire} : \hspace*{1cm}   On choisi $x$ le nombre de billes de Mathieu.\\

\textbf{\q}  \textbf{Traduire le problème par une équation.}\\



Pour cela, choisir une grandeur qui peut être exprimée de deux façons différentes.\\

Ici, il s'agit du nombre de billes qu'ils ont à eux 3.\hspace*{1cm} Il y a 53 billes en tout.\\

Mais cela peut aussi s'écrire : Mathieu + Pierre + Bryan = 53\\

On sait que :
\bi
\item \textcolor{green}{Mathieu = x}

\item Pierre a 2 billes de moins que Mathieu, à savoir \textcolor{blue}{Pierre = x - 2}

\item Bryan a 3 fois plus de billes que Mathieu, à savoir \textcolor{orange}{Bryan = 3x}\\
\ei

$\rightarrow$ L'équation à résoudre est donc : \hspace*{1cm} \textbf{\textcolor{green}{Mathieu} + \textcolor{blue}{Pierre} + \textcolor{orange}{Bryan} = 53}\\ 
 \hspace*{7.5cm}\textcolor{green}{ x} \hspace*{0.5cm} + \hspace*{0.1cm}\textcolor{blue}{(x - 2)} \hspace*{0.1cm} + \hspace*{0.1cm} \textcolor{orange}{3x} \hspace*{0.1cm} \textbf{= 53}\\

\textbf{\q}  \textbf{Résoudre l'équation:}

\begin{center}
$$x+ (x-2)+3x=53$$


$$x+ x-2+3x=53$$

$$5x - 2=53$$

$$5x - 2  \textcolor{red}{+2} = 53 \textcolor{red}{+2}$$

$$5x=55$$

$$\dfrac{5x}{\textcolor{red}{5}}=\dfrac{55}{\textcolor{red}{5}}$$

\fbox{$x = 11$}\\
\end{center}
\newpage
\vspace*{0.2cm}

\textbf{\q} \textbf{ Vérifier} que la solution de l'équation a du sens avec le problème concret.\\
     Ici, x étant un nombre de billes, la solution du problème doit être un nombre strictement positif.\\

   

\textbf{\q}  \textbf{Conclure. }  \\

On en conclut que :
\bi
\item \textcolor{green}{x = 11} \hspace*{0.75cm} Donc Mathieu possède 11 billes.

\item \textcolor{blue}{x - 2 = 11 - 2 = 9}\hspace*{0.75cm} Donc Pierre possède 9 billes.

\item\textcolor{orange}{$3x = 3 \times 11 = 33$ } \hspace*{0.75cm} Donc Bryan possède 33 billes.\\
\ei
