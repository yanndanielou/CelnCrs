\section{Calcul d'une expression sans parenthèses}

\subsection{Enchaînement d'additions et de soustraction}


\prop{
Dans une expression sans parenthèse comportant uniquement des additions et des soustractions, on  effectue les calculs de gauche à droite.
}

\ex{A = 58 - 7 + 3

A =   51   + 3

A = 54\\
}


\begin{Exoapp}
\initq

\q Calculer les expressions suivantes en indiquant les étapes de calculs :
\begin{multicols}{2}
A = 10 + 12 - 5 - 2 + 13\\
A = 22 - 5 - 2 + 13\\
A = 17 - 2 + 13\\
A = 15 + 13\\
A = 28 \\

\columnbreak

B = 15,3 - 5,3 + 12,4 - 1\\
B = 10 + 12,4 - 1\\
B = 22,4 - 1\\
B = 21,4\\

\end{multicols}
	
\q Calculer mentalement les expressions suivantes :
\begin{multicols}{2}
C = 12 + 5 - 3\\
C = 14\\

\columnbreak

D = 20 - 7 - 5 + 4\\
D = 12	\\

\end{multicols}

\end{Exoapp}


\begin{information}
Lorsqu'il n'y a que des additions, on peut effectuer les calculs dans l'ordre que l'on veut.
\end{information}

\ex {B = 25 + 9 + 75 \\
		B = 25 + 75 + 9\\
		B = 100 + 9\\
