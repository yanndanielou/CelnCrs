\textbf{\underline{{\large Mes objectifs:}}}

\begin{itemize}[label=$\hookrightarrow$]



\item Je dois connaître l'écriture décimale et utiliser la valeur des chiffres en fonction de leur rang dans l'écriture,

\end{itemize}
