\section{Rappels}

\subsection{Rappels sur la proportionnalité}

\df{
On dit que deux grandeurs sont \textbf{proportionnelles} lorsque l'on peut passer des valeurs de l'une aux valeurs de l'autre en multipliant par une même constante.\\
Cette constante est alors appelée \textbf{coefficient de proportionnalité}.

}\\

\ex

\begin{center}
\begin{tabular}{|l|c|c|c|c|}
	\hline
	\textbf{Nombre de chocolats} & 2 & 6 & 8 & 10 \\
	\hline
	\textbf{Prix (en \euro)} & 0,24 & 0,72 & 0,96 & 1,20\\
	\hline
\end{tabular}{$0,12$}
\medskip
\end{center}
Le tableau ci-dessus est un tableau de proportionnalité, le coefficient de proportionnalité est 0,12.\\

\rem

\begin{enumerate}
	\item On passe de la première la deuxième colonne en multipliant les valeurs par 3.
	\item La troisième colonne est la somme des deux précédentes.\\
\end{enumerate}

\subsection{Rappels sur les fonctions}

\df{
\textbf{Une fonction} est une application qui, à un nombre, fait correspondre un unique autre nombre.
On note f : x $ \mapsto$ f(x) et on lit : "f la fonction qui au nombre x associe le nombre f(x)".
}

\section{Fonctions linéaires}

\subsection{Définition}

\df{
On dit qu'une fonction $f$ est \textbf{linéaire} s'il existe un nombre $a$ tel que $f:x\longmapsto ax$.\\

Le nombre $a$ est appelé \textbf{coefficient directeur} ou \textbf{coefficient de linéarité} de la fonction $f$.
}\\


\ex

	\hline

	\hline

	\hline
	$i:x\longmapsto x$ & & \\
	\hline

\prop{
Soient $f$ une fonction linéaire telle que $f(x)=ax$ et $k$ un nombre.\\

Pour tous nombres $x_1$ et $x_2$ on a :
$$f(x_1+x_2)=f(x_1)+f(x_2)$$
$$f(k\times x_1)=k\times f(x_1)$$

}\\

\section{Fonctions affines}

\subsection{Définition}

\prop{
On dit qu'une fonction f est affine s'il existe deux nombres a
et b tel que f : $f:x\longmapsto ax + b$.\\
Le nombre \textbf{a} est appelé \textbf{coefficient directeur} de la fonction f et le nombre \textbf{b} est appelé \textbf{ordonnée à l'origine}.\\
}


\rem
\bi \item \textbf{Une fonction linéaire} est une fonction affine où b = 0.
\item \textbf{Une fonction constante} est une fonction affine où a = 0.\\
\ei

\ex
