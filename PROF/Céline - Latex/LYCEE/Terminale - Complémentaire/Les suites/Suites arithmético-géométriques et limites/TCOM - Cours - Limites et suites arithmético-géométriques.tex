 \textbf{\underline{{\large Activité d'introduction 1}}}\\
 
\textbf{ PARTIE A}\\
 On s'intéresse au nombre d'abonnés d'une plate-forme de streaming  de musique de France.\\
 En 2020, on compte 30 000 abonnés à la plate-forme. Chaque année, 90 \% des abonnés se réabonnent, et il y a 10 000 nouveaux abonnés.\\
 
 \noindent \initq \q Déterminer le nombre d'abonnés en 2021 et en 2022. \\
 \q On note $u_n$ le nombre d'abonnés en milliers en 2020 + $n$.\\
 \initqa \qa Exprimer $u_{n+1}$ en fonction de $u_n$. Calculer ensuite les valeurs $u_{40}$ et $u_{50}$  à l'aide de la calculatrice.\\

 \section{Limites de suites}

\dfTitre{Les suites convergentes}{

Une suite $u$ converge lorsqu'il existe un réel $\ell$ tel que tout intervalle ouvert I centré en $\ell$ contient tous les termes de la suite à partir d'un certain rang.\\

On écrit : \hspace*{0.5cm} $\limite{n}{+\infty}u_n=\ell$\\
}\\

 \dfTitre{Les suites divergentes}{
 
 Une suite $u$ diverge vers $+\infty$ (respectivement $-\infty$) si, pour tout réel $A>0$ (respectivement $B<0$), il existe un rang $p$ à partir duquel tous les termes de la suite sont plus grands que A (respectivement B).\\
 
 On écrit : \hspace*{0.5cm} $\limite{n}{+\infty}u_n=+\infty $ (respectivement $\limite{n}{+\infty}u_n=-\infty $) \\
 }\\

\end{center}

 \bi

\item Déterminer $\limite{n}{+\infty}\left( -\dfrac{2}{n} \right)$ et en déduire si la suite converge ou diverge.\\

\color{blue}

\textbf{ Par quotient,} $\lim\limits_{n \rightarrow + \infty} \dfrac{-2}{n}= 0$ \\

 \color{black}

\color{blue}

 Donc \textbf{ Par somme,} $\lim\limits_{n \rightarrow + \infty} \dfrac{1}{e^n}+n^2= +\infty$ \\

La suite $\left(  \dfrac{1}{e^n}+n^2\right)_{n\in \N^*}$ diverge.\\
 
 \color{black}

\color{blue}

Donc \textbf{ Par quotient,} $\lim\limits_{n \rightarrow + \infty} \dfrac{3-e^n}{1+e^{-n}}= -\infty$ \\

La suite $\left( \dfrac{3-e^n}{1+e^{-n}} \right)_{n\in \N^*}$ diverge.\\

  \prop{

 \bi
 
\item $\limite{n}{+\infty}u_n=+\infty$ alors  $\limite{n}{+\infty}v_n=+\infty$

\item $\limite{n}{+\infty}v_n=-\infty$ alors  $\limite{n}{+\infty}u_n=-\infty$

\ei  

  }
