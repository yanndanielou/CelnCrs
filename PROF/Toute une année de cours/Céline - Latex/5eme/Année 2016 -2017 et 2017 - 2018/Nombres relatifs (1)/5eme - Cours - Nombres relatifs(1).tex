\section{Vocabulaire}

\df{ Un nombre relatif est composé : 
\bi 
\item d'un signe : + ou - 
\item et d'une partie numérique
\ei
}\\

\rem \\

Si le signe est +, on dit que le nombre relatif est \textbf{positif}.\\

Si le signe est -, on dit que le nombre relatif est \textbf{négatif}.\\
\medskip

\ex

 50 ; 1 ; - 99 ; 0 ; - 7 ; + 22 ... sont des \textbf{entiers relatifs}. \\

6 ; 0,05 ; - 12,5 ; 0 ; - 0,8 ; + 4,9 ... sont des \textbf{décimaux relatifs}. \\

+ 19 ; + 310,2 ; + 1 ; 0 ; + 4,5 ; + 7 .................. sont \textbf{positifs}. \\

 - 2,6 ; - 43 ; - 2 ; 0 ; - 213,2 .................. sont \textbf{négatifs}.\\
\medskip


\textbf{Conventions} : 
\bi
\item Les nombres \textbf{positifs} pourront s'écrire \textbf{sans le signe +} . (les nombres négatifs gardent toujours leur signe - ). \

\item 0 est le seul nombre qui est \textbf{à la fois} un nombre \textbf{positif} et un nombre \textbf{négatif}.  0 n'a pas de signe.\\
\ei

+ 27 peut s'écrire 27. \\

+ 0,25 peut s'écrire 0,25 mais : - 12 $\ne$ 12\\

Activité 2 page 83\\

\df{
Chaque point d'une droite graduée est repéré par un nombre relatif appelé l'abscisse de ce point.}

\df{
On dit qu'une droite graduée est munie d'un repère (O,I) lorsque :
\bi
\item  O est l'origine du repère. L'abscisse de O est donc O(zéro). 
\item OI est l'unité de longueur que l'on reporte régulièrement à partir de O. 
On a donc : OI = 1. 
\item I est le point d'abscisse 1. On a donc fixé un sens.
\ei

\section{Comparaison des nombres relatifs}

\prop{
Tout nombre relatif positif est supérieur ou égal à 0.\\
Tout nombre relatif négatif est inférieur ou égal à 0.
}\\


\ex

\medskip


\prop{
Si deux nombres relatifs sont de signes contraires alors le plus petit est le nombre négatif.
}\\


\ex

Comparer -32 et 6,5 \\

\prop{
Si deux nombres relatifs sont négatifs alors le plus petit est celui qui a la plus grande partie numérique (ou la plus grande distance à zéro).
} \\

\ex 

Comparer - 37,2 et - 41,4\\

\noindent La partie numérique (ou distance à zéro) de - 37,2 est 37,2. \\
La partie numérique (ou distance à zéro) de - 41,4 est 41,4. \\
 \begin{center}
 41,4 > 37,2 \\
 Donc : - 41,4 < - 37,2\\
 \end{center}
