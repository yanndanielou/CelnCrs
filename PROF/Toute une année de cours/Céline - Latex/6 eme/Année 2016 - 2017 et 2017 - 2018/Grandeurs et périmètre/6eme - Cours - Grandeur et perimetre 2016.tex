Pour calculer le périmètre d'un polygone, on effectue la somme des mesures de ses côtés.

\subsection{Périmètre d'un cercle - Circonférence d'un cercle}

\df{
\textbf{La longueur d'un cercle} s'obtient en multipliant son diamètre d par le nombre $\pi$ (lire "pi"). D'où la formule suivante :\\
\begin{center}
$P= d \times \pi$ \hspace*{0.5cm} ou \hspace*{0.5cm}  $P=2 \times \pi \times r $
\end{center}
}\\

\ex

Donner le périmètre d'un cercle de rayon 3 cm.\\



\begin{Exoapp}

\initq
\q Donner la circonférence d'un cercle de diamètre 12 cm.\\
\reponse[3]\\

\q Donner le périmètre d'un cercle de rayon 5,5 cm.\\
\reponse[3]\\

\reponse[3]\\

\end{Exoapp}
